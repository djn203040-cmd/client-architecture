```typescript
import "server-only";
import { NextResponse, type NextRequest } from "next/server";
import { z } from "zod";
import { createClient } from "@/lib/supabase/server";
import { adminClient } from "@/lib/supabase/admin";

export const dynamic = "force-dynamic";

const MODE_A_CONFIRMATION_PHRASE = "send without review";

const BodySchema = z.object({
  mode: z.enum(["manual", "mode_a", "mode_b"]),
  confirmation_phrase: z.string().optional(),
});

export async function PATCH(req: NextRequest) {
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const parsed = BodySchema.safeParse(await req.json().catch(() => null));
  if (!parsed.success) {
    return NextResponse.json({ error: "Invalid input", details: parsed.error.issues }, { status: 400 });
  }
  const { mode, confirmation_phrase } = parsed.data;

  // High-friction server-side phrase enforcement for Mode A (DRAFT-009)
  if (mode === "mode_a") {
    if ((confirmation_phrase ?? "").trim() !== MODE_A_CONFIRMATION_PHRASE) {
      return NextResponse.json(
        { ok: false, reason: "phrase_mismatch" },
        { status: 400 },
      );
    }
  }

  const { error } = await adminClient
    .from("coaches")
    .update({ autonomous_mode: mode, updated_at: new Date().toISOString() })
    .eq("id", user.id);
  if (error) {
    return NextResponse.json({ ok: false, reason: `update_failed:${error.code ?? "unknown"}` }, { status: 500 });
  }

  // Audit log on Mode A toggle (operationally significant)
  if (mode === "mode_a") {
    await adminClient.from("activity_log").insert({
      coach_id: user.id,
      lead_id: null,
      event_type: "autonomous_mode_a_enabled",
      payload: { source: "settings_ui" },
    }).then(() => undefined).catch(() => undefined); // non-fatal — table may not exist; ignore
  }

  return NextResponse.json({ ok: true, mode });
}
```

```tsx
"use client";
import { useState, useTransition } from "react";
import { motion, AnimatePresence, useReducedMotion } from "framer-motion";
import { toast } from "sonner";
import { WarningCircle } from "@phosphor-icons/react";
import { RadioGroup, RadioGroupItem } from "@/components/ui/radio-group";
import { Alert, AlertDescription } from "@/components/ui/alert";
import { AutonomousModeAConfirmModal } from "./AutonomousModeAConfirmModal";

type Mode = "manual" | "mode_a" | "mode_b";

const OPTIONS: { value: Mode; label: string; description: string }[] = [
  {
    value: "manual",
    label: "Manual",
    description: "Every draft waits for your review. Recommended.",
  },
  {
    value: "mode_b",
    label: "Auto-send after 24h",
    description: "Drafts you don't act on send automatically when their scheduled time arrives. You still see them in the queue.",
  },
  {
    value: "mode_a",
    label: "Send without review",
    description: "Drafts skip the queue and send immediately. Not recommended.",
  },
];

const SUCCESS_COPY: Record<Mode, string> = {
  manual: "Manual review enabled.",
  mode_a: "Autonomous send enabled.",
  mode_b: "Auto-send after 24h enabled.",
};

export function AutonomousModeCard({ initialMode }: { initialMode: Mode }) {
  const [mode, setMode] = useState<Mode>(initialMode);
  const [pending, startTransition] = useTransition();
  const [modeAOpen, setModeAOpen] = useState(false);
  const [previousMode, setPreviousMode] = useState<Mode>(initialMode);
  const reduce = useReducedMotion();

  async function commit(next: Mode, confirmationPhrase?: string) {
    const r = await fetch("/api/settings/autonomous-mode", {
      method: "PATCH",
      headers: { "content-type": "application/json" },
      body: JSON.stringify({ mode: next, confirmation_phrase: confirmationPhrase }),
    });
    if (!r.ok) {
      toast.error("Couldn't save. Try again.");
      setMode(previousMode);
      return false;
    }
    toast.success(SUCCESS_COPY[next]);
    setPreviousMode(next);
    return true;
  }

  function handleChange(next: string) {
    const nextMode = next as Mode;
    setMode(nextMode); // optimistic select
    if (nextMode === "mode_a") {
      setModeAOpen(true);
      return;
    }
    startTransition(() => { void commit(nextMode); });
  }

  function onModeAConfirm(phrase: string) {
    setModeAOpen(false);
    startTransition(async () => {
      const ok = await commit("mode_a", phrase);
      if (!ok) setMode(previousMode);
    });
  }

  function onModeACancel() {
    setModeAOpen(false);
    setMode(previousMode); // revert silently
  }

  return (
    <>
      <div className="rounded-2xl backdrop-blur-md bg-white/10 dark:bg-white/5 border border-white/10 p-6 shadow-[inset_0_1px_0_rgba(255,255,255,0.08)] space-y-4">
        <h2 className="text-xl font-semibold">Autonomous mode</h2>
        <AnimatePresence>
          {mode === "mode_a" && (
            <motion.div
              initial={reduce ? { opacity: 0 } : { y: -8, opacity: 0 }}
              animate={reduce ? { opacity: 1 } : { y: 0, opacity: 1 }}
              exit={{ opacity: 0 }}
              transition={{ duration: 0.3 }}
            >
              <Alert className="bg-amber-100/70 dark:bg-amber-900/30 border-amber-300/40 text-amber-900 dark:text-amber-100">
                <WarningCircle className="size-3.5 mr-2" weight="regular" />
                <AlertDescription>
                  Autonomous send is active. Messages send without review.
                </AlertDescription>
              </Alert>
            </motion.div>
          )}
        </AnimatePresence>
        <RadioGroup value={mode} onValueChange={handleChange} disabled={pending} className="space-y-3">
          {OPTIONS.map((opt) => (
            <label
              key={opt.value}
              htmlFor={`mode-${opt.value}`}
              className={`block rounded-xl border p-4 cursor-pointer hover:bg-muted/30 transition-colors ${
                mode === opt.value ? "border-accent/40 ring-1 ring-accent/30" : "border-border"
              }`}
            >
              <div className="flex items-start gap-3">
                <RadioGroupItem value={opt.value} id={`mode-${opt.value}`} className="mt-0.5" />
                <div className="space-y-1">
                  <p className="text-sm font-medium text-foreground">{opt.label}</p>
                  <p className="text-sm text-muted-foreground">{opt.description}</p>
                </div>
              </div>
            </label>
          ))}
        </RadioGroup>
      </div>
      <AutonomousModeAConfirmModal
        open={modeAOpen}
        onConfirm={onModeAConfirm}
        onCancel={onModeACancel}
      />
    </>
  );
}
```

```tsx
"use client";
import { useState, useId, useEffect } from "react";
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter } from "@/components/ui/dialog";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";

const CONFIRMATION_PHRASE = "send without review";

interface Props {
  open: boolean;
  onConfirm: (phrase: string) => void;
  onCancel: () => void;
}

export function AutonomousModeAConfirmModal({ open, onConfirm, onCancel }: Props) {
  const [phrase, setPhrase] = useState("");
  const inputId = useId();
  const matches = phrase.trim() === CONFIRMATION_PHRASE; // case-sensitive

  // Reset on open
  useEffect(() => { if (open) setPhrase(""); }, [open]);

  return (
    <Dialog open={open} onOpenChange={(o) => { if (!o) onCancel(); }}>
      <DialogContent className="max-w-[460px] backdrop-blur-md bg-white/10 dark:bg-white/5 border border-white/10 shadow-[inset_0_1px_0_rgba(255,255,255,0.08)]">
        <DialogHeader>
          <DialogTitle className="text-xl font-semibold">Send without review?</DialogTitle>
        </DialogHeader>
        <p className="text-sm leading-[1.5] max-w-[65ch]">
          Drafts will send without your review. You won&apos;t see them in the queue before they go out. This setting is not recommended for most coaches.
        </p>
        <div className="space-y-2">
          <Label htmlFor={inputId} className="text-xs font-medium text-muted-foreground">
            Type the phrase to confirm
          </Label>
          <Input
            id={inputId}
            value={phrase}
            onChange={(e) => setPhrase(e.target.value)}
            placeholder={CONFIRMATION_PHRASE}
            autoComplete="off"
            autoCapitalize="off"
            spellCheck={false}
          />
        </div>
        <DialogFooter className="gap-2 sm:gap-2">
          <Button variant="ghost" onClick={onCancel} className="min-h-[44px]">
            Keep manual
          </Button>
          <Button
            variant="destructive"
            disabled={!matches}
            onClick={() => onConfirm(phrase.trim())}
            className="min-h-[44px]"
          >
            Enable autonomous send
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
}
```

```typescript
import "server-only";
import { inngest } from "@/inngest/client";
import { adminClient } from "@/lib/supabase/admin";
import { approveDraftAtomic } from "@/lib/drafts/approve-atomic";
import { runPreSendSafetyCheck } from "@/inngest/functions/sequence-step";

export const autonomousModeBTimer = inngest.createFunction(
  {
    id: "autonomous-mode-b-timer",
    name: "Autonomous Mode B — auto-send at scheduled time",
    cancelOn: [
      { event: "draft/approved_manually", if: "async.data.draftId == event.data.draftId" },
      { event: "draft/held_manually",      if: "async.data.draftId == event.data.draftId" },
      { event: "draft/cancelled",          if: "async.data.draftId == event.data.draftId" },
    ],
    retries: 2,
  },
  { event: "draft/created_mode_b" },
  async ({ event, step }) => {
    const { draftId, coachId, scheduledSendAt } = event.data as {
      draftId: string;
      coachId: string;
      scheduledSendAt: string;
    };

    await step.sleepUntil("sleep-until-send", new Date(scheduledSendAt));

    // 1) Re-check status — coach may have approved / held manually (cancelOn handles most cases, but belt-and-suspenders)
    const stillPending = await step.run("verify-still-pending", async () => {
      const { data } = await adminClient
        .from("drafts").select("status, lead_id, sequence_id")
        .eq("id", draftId).single();
      return data;
    });
    if (!stillPending || stillPending.status !== "pending") {
      return { cancelled: true, reason: `not_pending:${stillPending?.status ?? "missing"}` };
    }

    // 2) Pre-send safety check (Phase 3 D-25)
    const blocked = await step.run("safety-check", () =>
      runPreSendSafetyCheck(stillPending.lead_id, stillPending.sequence_id),
    );
    if (blocked) return { cancelled: true, reason: blocked };

    // 3) Advisory-lock CAS
    const result = await step.run("cas-approve", () =>
      approveDraftAtomic(draftId, "mode_b"),
    );
    if (!result.ok) return { cancelled: true, reason: result.reason };

    // 4) Fire Gmail send
    await step.sendEvent("send-via-gmail", {
      name: "draft/send_via_gmail",
      data: { draftId, coachId, source: "mode_b" },
    });

    return { sent: true, draftId };
  },
);
```
