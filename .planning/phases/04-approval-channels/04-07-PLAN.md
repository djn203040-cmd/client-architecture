**Create `apps/web/lib/notifications/channels/dashboard.ts`**:

```typescript
import "server-only";
import { randomUUID } from "crypto";
import { adminClient } from "@/lib/supabase/admin";
import type { TNotificationEvent, TChannelResult } from "@gsd/shared";

export async function sendDashboard(event: TNotificationEvent): Promise<TChannelResult> {
  const externalId = randomUUID();
  try {
    const { error } = await adminClient.from("notification_log").insert({
      coach_id: event.coachId,
      event_type: event.eventType,
      channel: "dashboard",
      external_id: externalId,
      status: "sent",
      payload: event.payload,
    });
    if (error) throw error;
    return { channel: "dashboard", status: "sent", external_id: externalId, error_message: null };
  } catch (err) {
    const msg = err instanceof Error ? err.message : "send_failed";
    return { channel: "dashboard", status: "failed", external_id: null, error_message: msg };
  }
}
```

> Note: Dashboard "send" is just the notification_log row. Realtime on `notification_log` filtered by `coach_id` (Phase 1 publication) delivers it to the dashboard.

**Create `apps/web/lib/notifications/dispatcher.ts`** — pure logic (no Inngest dep):

```typescript
import "server-only";
import { adminClient } from "@/lib/supabase/admin";
import type { TNotificationChannel, TNotificationEventType } from "@gsd/shared";

export interface EnabledChannelSet {
  dashboard: boolean;
  email: boolean;
  slack: boolean;
  whatsapp: boolean;
  sms: boolean;
}

// D-15: SMS is unconditional for hard bounce
const HARD_BOUNCE_UNCONDITIONAL: TNotificationChannel[] = ["sms"];
// D-13: Dashboard is always ON
const ALWAYS_ON: TNotificationChannel[] = ["dashboard"];

export async function computeEnabledChannels(
  coachId: string,
  eventType: TNotificationEventType,
): Promise<EnabledChannelSet> {
  const [prefsResult, integrationsResult] = await Promise.all([
    adminClient
      .from("notification_preferences")
      .select("channel, enabled")
      .eq("coach_id", coachId)
      .eq("event_type", eventType),
    adminClient
      .from("integrations")
      .select("provider, status")
      .eq("coach_id", coachId)
      .eq("status", "connected"),
  ]);
  const prefs = prefsResult.data ?? [];
  const connected = new Set((integrationsResult.data ?? []).map((i) => i.provider as string));

  // Coach is "connected" to dashboard implicitly; email is "connected" if any integration exists OR coach has email column.
  // Treat dashboard + email as always available; gate slack / whatsapp / sms behind provider connection.
  const channelConnected = (channel: TNotificationChannel): boolean => {
    if (channel === "dashboard" || channel === "email") return true;
    if (channel === "slack") return connected.has("slack");
    if (channel === "whatsapp" || channel === "sms") return connected.has("twilio");
    return false;
  };

  const prefMap = new Map<TNotificationChannel, boolean>();
  for (const row of prefs) prefMap.set(row.channel as TNotificationChannel, !!row.enabled);

  const isEnabled = (channel: TNotificationChannel): boolean => {
    if (ALWAYS_ON.includes(channel)) return true; // D-13 dashboard lock
    if (eventType === "hard_bounce" && HARD_BOUNCE_UNCONDITIONAL.includes(channel)) {
      // D-15: SMS forced ON for hard_bounce regardless of pref
      return channelConnected(channel);
    }
    const pref = prefMap.get(channel);
    if (pref === undefined) return false; // pref not seeded → off
    return pref && channelConnected(channel);
  };

  return {
    dashboard: isEnabled("dashboard"),
    email: isEnabled("email"),
    slack: isEnabled("slack"),
    whatsapp: isEnabled("whatsapp"),
    sms: isEnabled("sms"),
  };
}

export const CHANNELS_FOR_HARD_BOUNCE_UNCONDITIONAL = HARD_BOUNCE_UNCONDITIONAL;
```

**Create `apps/web/inngest/functions/notification-dispatcher.ts`** — register a dispatcher per event type for cleaner concurrency (one function trigger per event name; loop is allowed via Inngest's `triggers` array):

```typescript
import "server-only";
import { inngest } from "@/inngest/client";
import { computeEnabledChannels } from "@/lib/notifications/dispatcher";
import { sendDashboard } from "@/lib/notifications/channels/dashboard";
import { sendEmail } from "@/lib/notifications/channels/email";
import { sendSlack } from "@/lib/notifications/channels/slack";
import { sendWhatsApp } from "@/lib/notifications/channels/whatsapp";
import { sendSMS } from "@/lib/notifications/channels/sms";
import type { TNotificationEvent } from "@gsd/shared";

export const notificationDispatcher = inngest.createFunction(
  {
    id: "notification-dispatcher",
    name: "Notification dispatcher — fan out to enabled channels",
    retries: 2,
    concurrency: { key: "event.data.coachId", limit: 5 },
  },
  // Multiple event triggers — Inngest supports triggers array
  [
    { event: "notification/draft_ready" },
    { event: "notification/draft_followup" },
    { event: "notification/lead_replied" },
    { event: "notification/integration_broken" },
    { event: "notification/hard_bounce" },
  ],
  async ({ event, step }) => {
    const data = event.data as TNotificationEvent;
    const { coachId, eventType } = data;

    const enabled = await step.run("compute-enabled-channels", () =>
      computeEnabledChannels(coachId, eventType),
    );

    // Build promise array WITHOUT awaiting — Pitfall 8: distinct step ids
    const channelSteps = [
      enabled.dashboard && step.run("send-dashboard", () => sendDashboard(data)),
      enabled.email     && step.run("send-email",     () => sendEmail(data)),
      enabled.slack     && step.run("send-slack",     () => sendSlack(data)),
      enabled.whatsapp  && step.run("send-whatsapp",  () => sendWhatsApp(data)),
      enabled.sms       && step.run("send-sms",       () => sendSMS(data)),
    ].filter(Boolean) as Promise<unknown>[];

    const results = await Promise.allSettled(channelSteps);
    return {
      dispatched: results.length,
      results: results.map((r) => r.status === "fulfilled" ? { status: "fulfilled", value: r.value } : { status: "rejected" }),
    };
  },
);
```

**Create `apps/web/inngest/functions/draft-followup-cta.ts`** — single function with two sleeps (24h + 48h total) per assumption A8:

```typescript
import "server-only";
import { inngest } from "@/inngest/client";
import { adminClient } from "@/lib/supabase/admin";
import { holdDraftAtomic } from "@/lib/drafts/approve-atomic";

export const draftFollowupCta = inngest.createFunction(
  {
    id: "draft-followup-cta",
    name: "Draft follow-up CTA + HOLD cascade",
    cancelOn: [
      { event: "draft/approved_manually", if: "async.data.draftId == event.data.draftId" },
      { event: "draft/held_manually",      if: "async.data.draftId == event.data.draftId" },
      { event: "draft/cancelled",          if: "async.data.draftId == event.data.draftId" },
    ],
    retries: 2,
  },
  { event: "draft/created_pending" },
  async ({ event, step }) => {
    const { draftId, coachId, createdAt } = event.data as {
      draftId: string;
      coachId: string;
      createdAt: string;
    };

    // First sleep — 24h after draft created
    const firstWakeAt = new Date(new Date(createdAt).getTime() + 24 * 60 * 60 * 1000);
    await step.sleepUntil("sleep-24h-followup", firstWakeAt);

    // Check status still pending
    const firstCheck = await step.run("check-status-24h", async () => {
      const { data } = await adminClient
        .from("drafts").select("status, lead_id").eq("id", draftId).maybeSingle();
      return data;
    });
    if (!firstCheck || firstCheck.status !== "pending") {
      return { cancelled: true, stage: "24h", reason: `not_pending:${firstCheck?.status ?? "missing"}` };
    }

    // Increment followup_count + fire follow-up dispatcher event
    await step.run("increment-followup-count", async () => {
      await adminClient.rpc("increment_followup_count", { p_draft_id: draftId });
      // OR equivalent UPDATE drafts SET followup_count = followup_count + 1 WHERE id = ...
      // If no RPC exists, inline UPDATE — pick the simpler path during implementation
    });
    await step.run("fire-followup-notification", async () => {
      // Lookup payload bits for the followup template
      const { data: draft } = await adminClient
        .from("drafts").select("body, subject, scheduled_send_at, lead_id").eq("id", draftId).single();
      const { data: lead } = await adminClient
        .from("leads").select("name, email").eq("id", draft?.lead_id ?? "").maybeSingle();
      await inngest.send({
        name: "notification/draft_followup",
        data: {
          coachId,
          eventType: "draft_ready", // follow-up reuses same dispatcher; template picks followup variant via followup_count >= 1
          payload: {
            draftId,
            leadName: lead?.name ?? "your lead",
            sendTime: draft?.scheduled_send_at ?? "soon",
            subject: draft?.subject ?? "",
            body: draft?.body ?? "",
          },
        },
      });
    });

    // Second sleep — another 24h (so 48h total from draft creation)
    const secondWakeAt = new Date(firstWakeAt.getTime() + 24 * 60 * 60 * 1000);
    await step.sleepUntil("sleep-48h-cascade", secondWakeAt);

    const secondCheck = await step.run("check-status-48h", async () => {
      const { data } = await adminClient
        .from("drafts").select("status").eq("id", draftId).maybeSingle();
      return data;
    });
    if (!secondCheck || secondCheck.status !== "pending") {

  },
);
```

```tsx
  if (channel === "dashboard") return { locked: true, tooltip: "Dashboard notifications can't be turned off." };

    const lock = isLockedOn(eventType, channel);
    if (lock) return true;

                          Connect

                    </div>
                  </th>
                );
              })}

          </thead>

                <td className="text-sm font-medium text-foreground py-3 pr-4">{label}</td>
```

**Create `apps/web/app/api/settings/notifications/route.ts`** — single-cell PATCH with server-side lock enforcement:

```typescript
import "server-only";
import { NextResponse, type NextRequest } from "next/server";
import { z } from "zod";
import { createClient } from "@/lib/supabase/server";
import { adminClient } from "@/lib/supabase/admin";

export const dynamic = "force-dynamic";

const BodySchema = z.object({
  event_type: z.enum(["draft_ready", "lead_replied", "integration_broken", "hard_bounce"]),
  channel: z.enum(["dashboard", "email", "slack", "whatsapp", "sms"]),
  enabled: z.boolean(),
});

export async function PATCH(req: NextRequest) {
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const parsed = BodySchema.safeParse(await req.json().catch(() => null));
  if (!parsed.success) return NextResponse.json({ error: "Invalid input" }, { status: 400 });
  const { event_type, channel, enabled } = parsed.data;

  // D-13: Dashboard column is always ON
  if (channel === "dashboard" && !enabled) {
    return NextResponse.json({ ok: false, reason: "dashboard_always_on" }, { status: 400 });
  }
  // D-15: Hard bounce + SMS is locked ON
  if (event_type === "hard_bounce" && channel === "sms" && !enabled) {
    return NextResponse.json({ ok: false, reason: "hard_bounce_sms_always_on" }, { status: 400 });
  }

  const { error } = await adminClient.from("notification_preferences").upsert(
    {
      coach_id: user.id,
      event_type,
      channel,
      enabled,
      updated_at: new Date().toISOString(),
    },
    { onConflict: "coach_id,event_type,channel" },
  );
  if (error) {
    return NextResponse.json({ ok: false, reason: `update_failed:${error.code ?? "unknown"}` }, { status: 500 });
  }
  return NextResponse.json({ ok: true });
}
```
