**Step 2: Create `apps/web/lib/calendar/index.ts`.**

This module exports: (a) per-provider signature verification functions, (b) per-provider payload normalizers that return `TCalendarEvent | null`.

Key implementation rules:
- All verification functions use `createHmac` + `timingSafeEqual` from Node.js `crypto`
- Returns `false` (not throws) when header is missing or tampered
- Normalizers return `null` for unknown event types (silent ignore)
- Calendly: header is `calendly-webhook-signature`, value is `sha256=<hex>`, HMAC input is raw body
- Cal.com: header is `x-cal-signature-256`, value is hex digest (no prefix), HMAC input is `JSON.stringify(parsedPayload)`
- Acuity: header is `x-acuity-signature`, value is `base64(HMAC-SHA256(rawBody, apiKey))`
- Square: header is `x-square-hmacsha256-signature`, HMAC input is `notificationUrl + rawBody` (concatenation — NO separator)
- Setmore, MS Bookings, TidyCal: verification is a pass-through `() => true` with a comment: `// TODO: No documented signature verification for this provider — accept all`
- For Setmore/MS Bookings/TidyCal: `eventType` is always `"booking_created"` (no no-show support per CAL-009/D-08)

Key normalizer event type mapping:
- Calendly: `invitee_no_show.created` → `no_show`; `invitee.created` → `booking_created`
- Cal.com: `BOOKING_NO_SHOW_UPDATED` → `no_show` (manual mark only); `BOOKING_CREATED` → `booking_created`
- Acuity: `scheduled` → `booking_created` (no no_show mapping per CAL-009)
- Square: `booking.created` → `booking_created`; `booking.updated` → `rescheduled` (or `null` if unrelated)
- Setmore/MS Bookings/TidyCal: all events → `booking_created`

**Step 3: Create 7 calendar webhook routes.**

Each route follows this exact structure (from fireflies/route.ts canonical pattern):

```typescript
// apps/web/app/api/webhooks/calendar/calendly/route.ts
import "server-only";
import { adminClient } from "@/lib/supabase/admin";
import { inngest } from "@/inngest/client";
import { verifyCalendlySignature, normalizeCalendlyPayload } from "@/lib/calendar";
import { LEAD_NO_SHOW, LEAD_CALL_BOOKED, LEAD_CALL_COMPLETED } from "@client/shared/constants/events";

export async function POST(request: Request) {
  // 1. Read raw body FIRST — must happen before any JSON parsing
  const rawBody = await request.text();
  const coachId = new URL(request.url).searchParams.get("coachId");
  if (!coachId) return new Response("Missing coachId", { status: 400 });

  // 2. Signature verification — timing-safe (CAL-004, T-03-01)
  const valid = verifyCalendlySignature(
    rawBody,
    request.headers.get("calendly-webhook-signature"),
    process.env.CALENDLY_WEBHOOK_SECRET!
  );
  if (!valid) return new Response("Unauthorized", { status: 401 });

  // 3. Parse + normalize to TCalendarEvent
  const event = normalizeCalendlyPayload(JSON.parse(rawBody), coachId);
  if (!event) return new Response("OK", { status: 200 }); // unknown event type — ignore

  // 4. Deduplicate — calendar_events UNIQUE(provider, external_event_id) (SEQ-014, CAL-005)
  const { data: existing } = await adminClient
    .from("calendar_events")
    .select("id")
    .eq("provider", event.provider)
    .eq("external_event_id", event.externalEventId)
    .maybeSingle();
  if (existing) return new Response("OK", { status: 200 }); // silent dedup

  // 5. Find lead by email
  let leadId: string | null = null;
  if (event.leadEmail) {
    const { data: lead } = await adminClient
      .from("leads")
      .select("id")
      .eq("coach_id", coachId)
      .eq("email", event.leadEmail)
      .maybeSingle();
    leadId = lead?.id ?? null;
  }

  // 6. Insert calendar_events row
  await adminClient.from("calendar_events").insert({
    coach_id: coachId,
    provider: event.provider,
    external_event_id: event.externalEventId,
    lead_id: leadId,
    event_type: event.eventType,
    payload: event.rawPayload,
  });

  // 7. Fire Inngest event with deterministic ID (SEQ-007 — prevents duplicate function starts on webhook retries)
  const inngestEventName =
    event.eventType === "no_show" ? LEAD_NO_SHOW :
    event.eventType === "booking_created" ? LEAD_CALL_BOOKED :
    null;

  if (inngestEventName && leadId) {
    await inngest.send({
      id: `${event.provider}-${event.externalEventId}`, // deterministic ID
      name: inngestEventName,
      data: {
        coachId,
        leadId,
        provider: event.provider,
        externalEventId: event.externalEventId,
        eventStartAt: event.eventStartAt,
        eventEndAt: event.eventEndAt,
      },
    });
  }

  return new Response("OK", { status: 200 });
}
```
