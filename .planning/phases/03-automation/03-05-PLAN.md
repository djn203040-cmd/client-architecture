**Create `apps/web/lib/unsubscribe-token.ts`** — HMAC-signed token for CAN-SPAM compliance:

```typescript
import "server-only";
import { createHmac, timingSafeEqual } from "crypto";

interface UnsubscribePayload {
  leadId: string;
  coachId: string;
  t: number; // timestamp — not a strict expiry, but useful for audit
}

/**
 * Generates an HMAC-signed unsubscribe token.
 * Token format: base64url(JSON(payload)).hexHmac
 * COMPLY-001, COMPLY-002, T-03-04
 */
export function generateUnsubscribeToken(leadId: string, coachId: string): string {
  const secret = process.env.UNSUBSCRIBE_SECRET;
  if (!secret) throw new Error("UNSUBSCRIBE_SECRET env var not set");

  const payload: UnsubscribePayload = { leadId, coachId, t: Date.now() };
  const encodedPayload = Buffer.from(JSON.stringify(payload)).toString("base64url");
  const hmac = createHmac("sha256", secret).update(encodedPayload).digest("hex");

  return `${encodedPayload}.${hmac}`;
}

/**
 * Verifies an unsubscribe token and returns the payload if valid.
 * Uses timing-safe comparison to prevent timing oracle attacks (T-03-04).
 * Returns null if token is missing, malformed, or signature invalid.
 */
export function verifyUnsubscribeToken(
  token: string
): UnsubscribePayload | null {
  const secret = process.env.UNSUBSCRIBE_SECRET;
  if (!secret) return null;

  const parts = token.split(".");
  if (parts.length !== 2) return null;

  const [encodedPayload, providedHmac] = parts;

  // Recompute expected HMAC
  const expectedHmac = createHmac("sha256", secret)
    .update(encodedPayload)
    .digest("hex");

  // Timing-safe comparison (T-03-04)
  const a = Buffer.from(providedHmac);
  const b = Buffer.from(expectedHmac);
  if (a.length !== b.length) return null;
  if (!timingSafeEqual(a, b)) return null;

  try {
    return JSON.parse(
      Buffer.from(encodedPayload, "base64url").toString("utf8")
    ) as UnsubscribePayload;
  } catch {
    return null;
  }
}

/**
 * Builds the full unsubscribe URL for embedding in email footers.
 * COMPLY-001: must be in every outbound email.
 */
export function buildUnsubscribeUrl(leadId: string, coachId: string): string {
  const token = generateUnsubscribeToken(leadId, coachId);
  const appUrl = process.env.NEXT_PUBLIC_APP_URL ?? "";
  return `${appUrl}/api/unsubscribe?token=${encodeURIComponent(token)}`;
}
```

**Create `apps/web/app/api/unsubscribe/route.ts`** — no-auth endpoint (CAN-SPAM link must work without login):

```typescript
import "server-only";
import { adminClient } from "@/lib/supabase/admin";
import { inngest } from "@/inngest/client";
import { verifyUnsubscribeToken } from "@/lib/unsubscribe-token";
import { LEAD_UNSUBSCRIBED } from "@client/shared/constants/events";

export const dynamic = "force-dynamic";

export async function GET(request: Request) {
  const token = new URL(request.url).searchParams.get("token");

  if (!token) {
    return new Response(null, {
      status: 302,
      headers: { Location: "/unsubscribe?error=invalid_token" },
    });
  }

  // Verify HMAC-signed token (T-03-04)
  const payload = verifyUnsubscribeToken(token);
  if (!payload) {
    return new Response(null, {
      status: 302,
      headers: { Location: "/unsubscribe?error=invalid_token" },
    });
  }

  const { leadId, coachId } = payload;

  // Verify lead belongs to this coach before making any changes
  const { data: lead } = await adminClient
    .from("leads")
    .select("id, status")
    .eq("id", leadId)
    .eq("coach_id", coachId)
    .maybeSingle();

  if (!lead) {
    return new Response(null, {
      status: 302,
      headers: { Location: "/unsubscribe?error=not_found" },
    });
  }

  // Idempotent: if already unsubscribed, still show success page
  if (lead.status === "unsubscribed") {
    return new Response(null, {
      status: 302,
      headers: { Location: "/unsubscribe?done=1" },
    });
  }

  // D-19: Update lead, sequence, and drafts (multi-table)
  await adminClient
    .from("leads")
    .update({ status: "unsubscribed", do_not_contact: true })
    .eq("id", leadId);

  await adminClient
    .from("sequences")
    .update({ status: "cancelled" })
    .eq("lead_id", leadId)
    .eq("coach_id", coachId)
    .in("status", ["active", "paused"]);

  await adminClient
    .from("drafts")
    .update({ status: "cancelled" })
    .eq("lead_id", leadId)
    .eq("coach_id", coachId)
    .eq("status", "pending");

  // Log to activity timeline
  await adminClient.from("lead_events").insert({
    lead_id: leadId,
    coach_id: coachId,
    event_type: "state_changed",
    payload: { to: "unsubscribed", trigger: "unsubscribe_link" },
    triggered_by: "system",
  });

  // Fire Inngest cancelOn event so any sleeping sequences also terminate (SEQ-008)
  await inngest.send({
    name: LEAD_UNSUBSCRIBED,
    data: { coachId, leadId },
  });

  return new Response(null, {
    status: 302,
    headers: { Location: "/unsubscribe?done=1" },
  });
}
```

**Create `apps/web/inngest/functions/bounce-handler.ts`** (COMPLY-005/007, STATE-008):

The LEAD_BOUNCED event is fired by `gmail-monitor.ts` when a MAILER-DAEMON message is detected. The payload includes `coachId`, `messageId` (the MAILER-DAEMON message), and optional `subject` and `fromAddress`.

Actions (D-20):
1. Fetch the bounce message from Gmail to extract the bounced email address
2. Find the lead by email address
3. Set `lead.bounced = true`
4. Cancel active sequences for this lead
5. Log to lead_events
6. Insert coach notification row (Phase 4 delivers the notification; Phase 3 queues it)

COMPLY-008: Soft bounces (transient) do NOT pause sequences — only hard bounces. The gmail-monitor already filters: only MAILER-DAEMON emails fire LEAD_BOUNCED. So all events reaching this handler are hard bounces.

```typescript
import { inngest } from "@/inngest/client";
import { adminClient } from "@/lib/supabase/admin";
import { LEAD_BOUNCED } from "@client/shared/constants/events";
import { getGmailClientForCoach } from "@/lib/gmail/client";
import { extractBouncedEmail } from "@/lib/gmail/bounce-detector";
import { extractHeader } from "@/lib/gmail/thread";

export const bounceHandler = inngest.createFunction(
  { id: "bounce-handler" },
  { event: LEAD_BOUNCED },
  async ({ event, step }) => {
    const { coachId, messageId, subject: subjectFromEvent } = event.data as {
      coachId: string;
      messageId: string;
      subject?: string;
      fromAddress?: string;
    };

    // Step 1: Fetch bounce message to extract bounced email address
    const bouncedEmail = await step.run("extract-bounced-email", async () => {
      try {
        const gmail = await getGmailClientForCoach(coachId);
        const msg = await gmail.users.messages.get({
          userId: "me",
          id: messageId,
          format: "metadata",
          metadataHeaders: ["Subject", "From"],
        });

        const headers = msg.data.payload?.headers ?? [];
        const subject = extractHeader(headers, "subject");
        const snippet = msg.data.snippet ?? "";

        return extractBouncedEmail(subject || subjectFromEvent || "", snippet);
      } catch {
        // If we can't fetch the message, we can't identify the lead
        return null;
      }
    });

    if (!bouncedEmail) {
      // COMPLY-008: Cannot identify which lead bounced — log and exit
      return { ok: false, reason: "could_not_extract_email" };
    }

    // Step 2: Find lead by bounced email address
    const lead = await step.run("find-lead", async () => {
      const { data } = await adminClient
        .from("leads")
        .select("id, status, email, name")
        .eq("coach_id", coachId)
        .eq("email", bouncedEmail)
        .maybeSingle();
      return data;
    });

    if (!lead) {
      return { ok: false, reason: "lead_not_found", bouncedEmail };
    }

    // Step 3: Mark lead as bounced (COMPLY-007, STATE-008)
    await step.run("mark-lead-bounced", async () => {
      await adminClient
        .from("leads")
        .update({ bounced: true })
        .eq("id", lead.id);

      // Cancel active and paused sequences (D-20)
      await adminClient
        .from("sequences")
        .update({ status: "cancelled" })
        .eq("lead_id", lead.id)
        .eq("coach_id", coachId)
        .in("status", ["active", "paused"]);

      // Log to activity timeline
      await adminClient.from("lead_events").insert({
        lead_id: lead.id,
        coach_id: coachId,
        event_type: "email_bounced",
        payload: {
          bouncedEmail,
          messageId,
          reason: "hard_bounce",
        },
        triggered_by: "system",
      });
    });

    // Step 4: Insert pending_notification row — queued for multi-channel delivery when Phase 4
    // notification channels are configured. Phase 4 (NOTIFY-001–008) owns delivery.
    await step.run("queue-bounce-notification", async () => {
      await adminClient.from("notification_log").insert({
        coach_id: coachId,
        event_type: "lead_bounced",
        status: "pending",
        // Include lead info in external_id field for Phase 4 to use
        external_id: JSON.stringify({
          leadName: lead.name,
          leadEmail: bouncedEmail,
          leadId: lead.id,
          cta: "Update the email address in the lead profile to resume outreach.",
        }),
      });
    });

    return {
      ok: true,
      leadId: lead.id,
      bouncedEmail,
      sequencesCancelled: true,
      notificationQueued: true,
    };
  }
);
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-18 | Tampering | /api/unsubscribe token forgery | mitigate | Token is HMAC-SHA256 signed with UNSUBSCRIBE_SECRET env var. timingSafeEqual comparison prevents timing oracle attacks. Payload contains leadId + coachId — forging requires knowledge of the secret. Lead ownership verified against DB before any update. |
| T-03-19 | Spoofing | /api/unsubscribe — no session required | accept | Unsubscribe must work without login (CAN-SPAM requirement). The signed token provides sufficient assurance. The worst case of a successful forgery is unsubscribing a lead — a privacy-protective outcome. |
| T-03-20 | Elevation of Privilege | Unsubscribe affects other coaches' leads | mitigate | Token payload includes coachId. DB update includes `.eq("coach_id", coachId)` — cross-coach forgery impossible even with a valid token for a different coach. |
| T-03-21 | Tampering | Bounce handler processes fake LEAD_BOUNCED events | accept | LEAD_BOUNCED is fired by our gmail-monitor after detecting MAILER-DAEMON in the coach's Gmail inbox. Only our internal Inngest environment can fire this event. Additionally, bounce handler looks up the bounced email address from Gmail directly — a fake event with no matching message causes graceful exit. |
